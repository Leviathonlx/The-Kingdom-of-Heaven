	is_male = yes
	is_ruler = no

		place_in_line_of_succession = {
		  target = PREV
		  value > 2
		}
	}
}

		has_trait = lustful

		has_trait = greedy
		has_trait = lazy
		has_trait = wrathful
		has_trait = callous 
		has_trait = arrogant 
	}
}

	OR = {
		has_trait = just
		has_trait = compassionate 
		has_trait = generous
		has_trait = patient
	}
}

is_pious_trigger = {
	has_pious_trait_trigger = yes
	has_impious_trait_trigger = no
}

	}

	}
}
