is_inaccessible_trigger = { #Remember to write this "is_inaccessible_trigger = yea" when scripting, or the tooltip will be messed up (it reads "Is NOT in Hiding, in Seclusion, or out traveling")
	is_adult = yes
	OR = {
		is_ai = no
		is_busy_in_events = no
	}
	is_in_an_activity = no
	is_at_war = no
	is_imprisoned = no
	is_healthy = yes
	NOR = {
		exists = var:booked_for_a_party
		exists = var:engaged_in_single_combat
	}
}

		has_trait = incapable
	}
