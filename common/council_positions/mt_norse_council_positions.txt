councillor_master_of_the_ships = {
	name = councillor_master_of_the_ships
	skill = martial
	tooltip = game_concept_master_of_the_ships_desc
	auto_fill = 
	{
		highest_held_title_tier < tier_duchy
	}

	valid_position = 
	{
		highest_held_title_tier >= tier_duchy
		has_culture_group = culture_group:north_germanic_group
	}

	valid_character = {
		exists = root.liege_or_court_owner
		can_be_master_of_the_ships_trigger = { COURT_OWNER = root.liege_or_court_owner }
	}

	modifier = {
		name = councillor_master_of_the_ships_modifier
		prowess = 1
		monthly_prestige = 0.5
		monthly_martial_lifestyle_xp_gain_mult = 0.01
	}	

	on_get_position = {
		gave_minor_title_effect = yes 					
	}
	
	on_fired_from_position = {
		if = {
			limit = { is_powerful_vassal = no }
			remove_opinion = 
			{
				modifier = gave_minor_title_normal_vassal_opinion
				target = liege
			}
		}
	else_if = {
			remove_opinion = 
			{
				modifier = gave_minor_title_powerful_vassal_opinion
				target = liege
			}
		}
	}	
	
	portrait_animation = master_of_the_horses
}
